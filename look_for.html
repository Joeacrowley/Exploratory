2.3 Display look_for

tester <- look_for(df, details = FALSE)

look_h <- function(data, 
                   print = "h", ## set to f for flextable
                   capt = NULL,  ## Set to string to use as caption. 
                   export = NULL ## Set to file location
                   ){ 
  
  ht <- 
    
    data |>
  
    huxtable::as_hux() %>%
    
    ## Applied to all cells
    ## amount of space on each side of a cell. 
    huxtable::set_all_padding(1) |> 
    ## amount of space on each side of a cell. 
    huxtable::set_outer_padding(1) |>
    ## decimal place display (works on all numbers found in a cell, even non-numeric data)
    huxtable::set_number_format(2) |> 
    ## Font size
    huxtable::set_font_size(8) |> 
    ## Font
    huxtable::set_font("Arial") |>
    
    ## Set zero decimal places
    huxtable::set_number_format(everywhere, value = fmt_pretty(0)) %>%
    
    ## Heading formatting
    ## bold top row (the column names)
    huxtable::set_bold(row = 1, col = everywhere) |> 
    ## Add bottom border of row 1, default value = 0.4 points.
    huxtable::set_bottom_border(row = 1, col = everywhere) |>  
  
    ## Right hand border to variable column
    huxtable::set_right_border(everywhere, ## all rows
                     2,  ## column 2 (variable name)
                     brdr(2, "solid", "grey")) |> 
    
    ## Set NA string
    huxtable::set_na_string("-") |>
  
    ## External borders only... 
    huxtable::set_outer_borders(everywhere, everywhere, brdr(1,style = "solid", "black")) |>
    
    ## Background colour in even rows to grey, all columns
    huxtable::set_background_color(evens, everywhere, "grey95") |>
    
    ## Right align first column
    huxtable::set_align(everywhere, everywhere, "left") |>
    
    ## Wrap columns (only works when width set to (default is TRUE anyway))
    huxtable::set_wrap(everywhere, everywhere, TRUE) |>
    
    ## Width of the table as a proportion of the available space - here 100%
    huxtable::set_width(1)
    
    ## Add table caption
    if(!is.null(capt)){ht <- ht |> huxtable::set_caption(capt)} 
    
    ## Control column widths, dynamic based on content
    col_lengths <- apply(ht, 2, function(col) max(nchar(as.character(col))))
    relative_widths <- col_lengths / sum(col_lengths)
    huxtable::col_width(ht) <- relative_widths
    
    ## Export to Excel? 
    if (!is.null(export)){
        
        wb <- huxtable::as_Workbook(ht = ht, sheet = "lookfor")
        ## Excel cell locations
        end_row <- 1 + nrow(data) ## 1 here because huxtable adds header row to number of rows. 
        end_col <- ncol(data)
        ## Required to remove row heights set by as_Workbook() - these are problematic
        removeRowHeights(wb, "lookfor", rows = 2:end_row) 
        
        addStyle(wb, 
             "lookfor", 
             style = createStyle(wrapText = T), 
             cols = 2:end_col, 
             rows = 2:end_row, 
             gridExpand = TRUE, 
             stack = T)
    
        ## Remove huxtable defined column widths and apply openxlsx standards
        removeColWidths(wb, "lookfor", cols = 2:end_col)
        
        ## Temporary function to identify column width
        .col_width <- function(huxt) { 
      
            ## Extract first row of huxtable
            header_row_char_length <- 1:ncol(huxt) %>% map(., ~as_tibble(huxt[,.x])) %>% map_int(., ~max(str_length(as_vector(.x))))
          
            ## Multiple by number of pixels per character
            header_proposed_width <- header_row_char_length + 1
          
            ## Cut off of 30 imposed, and lower bound of 10. 
            header_proposed_width_cut_off <- ifelse(header_proposed_width > 50, 50, header_proposed_width)
            header_proposed_width_cut_off <- ifelse(header_proposed_width_cut_off < 10, 10, header_proposed_width_cut_off)
            
          return(header_proposed_width_cut_off)
          
          }
        
        col_widths <- ht %>% .col_width()
        setColWidths(wb, "lookfor", cols = 1:end_col, widths = col_widths)
        saveWorkbook(wb, file = export, overwrite = TRUE)
    }
    
    
    if(print == "f"){ht <- ht %>% huxtable::as_flextable() %>% flextable::autofit()}
    
    return(ht)
    
}

2.4 look_for with a search filter

look_for2 <- function(data, 
                      filter, 
                      details = "med", ## high or low
                      form = F,  ## if T then format
                      type = "f" ## set "h" for huxtable output
                      ){
  
  results <-
      look_for(data) %>%
      convert_list_columns_to_character() %>%
      filter(grepl(filter, variable, ignore.case = T) |
             grepl(filter, label, ignore.case = T) |
             grepl(filter, levels, ignore.case = T) |
             grepl(filter, value_labels, ignore.case = T)
             ) 
  
  if(details == "med"){
    results <- results %>% select(pos, variable, label, levels, value_labels)
  } else if (details == "low") {
    results <- results %>% select(pos, variable, label)
  } else if (details == "high") {
    results <- results %>% select(pos, variable, label)
  }
  
  if(form == T){
    results <- results %>% look_h(print = type, capt = paste0("Search for: ", filter))
  }
    
  return(results)
  
}
